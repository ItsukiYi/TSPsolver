% ============================================================
% TABLE 7: Five Improvements Summary
% ============================================================
\begin{table}[h]
\centering
\caption{Five Improvements Attempted on DualOpt.}
\label{tab:improvements}
\small
\begin{tabular}{@{}clll@{}}
\toprule
\textbf{\#} & \textbf{Improvement} & \textbf{Strategy} & \textbf{Outcome} \\
\midrule
1 & Heatmap-Guided Reviser   & Skip revision on confident windows    & No effect ($\Delta\!=\!0.00\%$) \\
2 & \textbf{DIFUSCO$\rightarrow$DualOpt} & \textbf{Replace initial tour with diffusion} & \textbf{\textcolor{bestgreen}{SUCCESS ($-$1.67\%)}} \\
3 & Adaptive Window Sizing   & 2-opt diagnostic $\rightarrow$ iteration budget & No improvement \\
4 & Fragment Freezing        & Lock edges where two solvers agree     & Mixed: 4/10 improved, unstable \\
5 & Destroy-and-Repair       & Remove low-conf.\ edges, greedy reconnect  & Degradation \\
\bottomrule
\end{tabular}

\vspace{2pt}
{\footnotesize \textbf{Insight:} All heatmap-guided attempts (\#1,3,4,5) failed because TSP-50 is ``solved'' for DualOpt's reviser.}\\
{\footnotesize Only \#2 succeeded — it changes the \textit{input} (initial tour), not the reviser's \textit{internal process}.}
\end{table}
